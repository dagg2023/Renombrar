34
35/**
36 * @class Rename
37 * @brief Clase principal que inicia la aplicación JavaFX.
38 *
39 * Configura:
40 * - La ventana principal
41 * - La escena con el FXML
42 * - El icono de la aplicación
43 */
44
45class Rename : Application() {
46 /**
47 * @fn start
48 * @brief Método principal de JavaFX.
49 * @param stage Ventana principal proporcionada por JavaFX.
50 */
51
52 override fun start(stage: Stage) {
53 val fxmlLoader = FXMLLoader(Rename::class.java.getResource("renamer_view.fxml"))
54 val scene = Scene(fxmlLoader.load(), 400.0, 650.0)
55
56 stage.title = "[ R E N O M B R A R ]"
57 val icon = Image(javaClass.getResourceAsStream("/app_icon.png"))
58 stage.icons.add(icon)
59 stage.scene = scene
60 stage.show()
61 }
62}
